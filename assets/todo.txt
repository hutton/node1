var email = require('./incoming.json');
var _ = require("./underscore.js");

function processIncomingMandrillEmail(email){
	if (email.event != "inbound"){
		return null;
	}

	var parsed = {};

	parsed.to = email.msg.to[0][0];
	parsed.fromName = email.msg.from_name;
	parsed.from = email.msg.from_email;
	parsed.subject = email.msg.subject;

	if (_.has(email.msg, "text") && email.msg.text !== null){
		parsed.message = email.msg.text;
	} else if (_.has(email.msg, "html") && email.msg.html !== null){
		parsed.message = email.msg.text;
	}

	return parsed;
}
